## 一、架构总览

### 1.1 架构原则

| 原则 | 说明 |
| --- | --- |
| **关注点分离** | 渲染、逻辑、输入、数据各司其职 |
| **实体-系统分离** | 数据放实体（Entity），行为放系统（System） |
| **单一入口** | 所有逻辑从 `main.ts` 启动 |
| **可扩展性** | 新增功能不破坏现有结构（V2 容易扩展） |
| **零全局变量** | 所有状态封装在类实例中 |

### 1.2 分层架构

```
┌─────────────────────────────────────┐
│ 表现层 Presentation                 │
│   ├─ Renderer（Canvas 绘制）        │
```

### 3.2 核心引擎层（core/）

#### `core/Game.ts` —— 游戏主类 ⭐⭐⭐

**职责**：
- 游戏主循环（Fixed Timestep）
- 状态机管理
- 协调各系统调用

```ts
import { CONFIG } from '../config';
import { Renderer } from './Renderer';
import { Input } from './Input';
import { Snake } from '../entities/Snake';
import { Food } from '../entities/Food';
import { CollisionSystem } from '../systems/CollisionSystem';
import { ScoreSystem } from '../systems/ScoreSystem';
import { AudioSystem } from '../systems/AudioSystem';
import { storage } from '../utils/storage';

enum GameState {
  MENU = 'menu',
  PLAYING = 'playing',
  PAUSED = 'paused',
  GAME_OVER = 'game_over',
}

interface GameDeps {
  canvas: HTMLCanvasElement;
  scoreEl: HTMLElement;
  bestEl: HTMLElement;
  onScoreUpdate: (score: number, best: number) => void;
}

export class Game {
  private state: GameState = GameState.MENU;
  private lastTime = 0;
  private accumulator = 0;
  private readonly step = 1000 / 60; // 60Hz 逻辑更新
  private rafId = 0;

  private snake!: Snake;
  private food!: Food;
  private renderer!: Renderer;
  private input!: Input;
  private collision!: CollisionSystem;
  private score!: ScoreSystem;
  private audio!: AudioSystem;

  private bestScore = 0;

  constructor(private deps: GameDeps) {
    this.init();
  }

  private init() {
    // 读取最高分
    this.bestScore = storage.get<number>(CONFIG.STORAGE_KEY, 0);

    // 创建实体
    this.snake = new Snake();
    this.food = new Food();
    this.food.spawn(this.snake);

    // 创建系统
    this.renderer = new Renderer(this.deps.canvas);
    this.input = new Input(this.deps.canvas);
    this.collision = new CollisionSystem();
    this.score = new ScoreSystem(this.deps.onScoreUpdate);
    this.audio = new AudioSystem();

    // 绑定输入
    this.input.onDirection = (dir) => this.snake.queueDirection(dir);
    this.input.onPause = () => this.togglePause();
    this.input.onConfirm = () => this.handleConfirm();
    this.input.onReset = () => this.reset();
  }

  public start() {
    this.lastTime = performance.now();
    this.rafId = requestAnimationFrame((t) => this.loop(t));
  }

  public stop() {
    cancelAnimationFrame(this.rafId);
  }

  private loop(timestamp: number) {
    const delta = timestamp - this.lastTime;
    this.lastTime = timestamp;
    this.accumulator += delta;

    while (this.accumulator >= this.step) {
      this.update(this.step);
      this.accumulator -= this.step;
    }

    this.render();
    this.rafId = requestAnimationFrame((t) => this.loop(t));
  }

  private update(_dt: number) {
    if (this.state !== GameState.PLAYING) return;

    const moveInterval = this.snake.moveInterval;
    this.snake.tickAccumulator = (this.snake.tickAccumulator ?? 0) + _dt;

    while (this.snake.tickAccumulator >= moveInterval) {
      this.snake.tickAccumulator -= moveInterval;
      this.tickSnake();
    }
  }

  private tickSnake() {
    const ate = this.snake.move(this.food.position);
    if (ate) {
      this.score.add(CONFIG.SCORE.PER_FOOD);
      this.audio.playEat();
      this.food.spawn(this.snake);

      // 难度递增
      this.snake.moveInterval = Math.max(
        CONFIG.SPEED.MIN_MS,
        this.snake.moveInterval * CONFIG.SPEED.ACCEL
      );
    }

    if (this.collision.checkWall(this.snake) ||
        this.collision.checkSelf(this.snake)) {
      this.gameOver();
    }
  }

  private render() {
    this.renderer.clear();
    this.renderer.drawGrid();
    this.renderer.drawFood(this.food);
    this.renderer.drawSnake(this.snake);
    this.renderer.drawOverlay(this.state);
  }

  // 状态切换
  private setState(s: GameState) {
    this.state = s;
  }

  private togglePause() {
    if (this.state === GameState.PLAYING) this.setState(GameState.PAUSED);
    else if (this.state === GameState.PAUSED) this.setState(GameState.PLAYING);
  }

  private handleConfirm() {
    if (this.state === GameState.MENU) this.startGame();
    else if (this.state === GameState.GAME_OVER) this.startGame();
  }

  private startGame() {
    this.snake.reset();
    this.food.spawn(this.snake);
    this.score.reset();
    this.snake.moveInterval = CONFIG.SPEED.INITIAL_MS;
    this.setState(GameState.PLAYING);
  }

  private gameOver() {
    this.audio.playDeath();
    const final = this.score.current;
    if (final > this.bestScore) {
      this.bestScore = final;
      storage.set(CONFIG.STORAGE_KEY, final);
    }
    this.setState(GameState.GAME_OVER);
  }

  private reset() {
    this.startGame();
  }
}
```

#### `core/Renderer.ts` —— 渲染封装

**职责**：
- 封装 Canvas 2D 绘制 API
- 提供高层 draw 方法

```ts
import { CONFIG } from '../config';
import { Snake } from '../entities/Snake';
import { Food } from '../entities/Food';

type GameState = 'menu' | 'playing' | 'paused' | 'game_over';

export class Renderer {
  private ctx: CanvasRenderingContext2D;

  constructor(private canvas: HTMLCanvasElement) {
    const ctx = canvas.getContext('2d');
    if (!ctx) throw new Error('Canvas 2D not supported');
    this.ctx = ctx;
    canvas.width = CONFIG.CANVAS.WIDTH;
    canvas.height = CONFIG.CANVAS.HEIGHT;
  }

  clear() {
    this.ctx.fillStyle = CONFIG.COLORS.BG;
    this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);
  }

  drawGrid() {
    this.ctx.strokeStyle = CONFIG.COLORS.GRID_LINE;
    this.ctx.lineWidth = 0.5;
    const { COLS, ROWS, CELL_SIZE } = CONFIG.GRID;
    for (let i = 1; i < COLS; i++) {
      this.ctx.beginPath();
      this.ctx.moveTo(i * CELL_SIZE, 0);
      this.ctx.lineTo(i * CELL_SIZE, ROWS * CELL_SIZE);
      this.ctx.stroke();
    }
    for (let i = 1; i < ROWS; i++) {
      this.ctx.beginPath();
      this.ctx.moveTo(0, i * CELL_SIZE);
      this.ctx.lineTo(COLS * CELL_SIZE, i * CELL_SIZE);
      this.ctx.stroke();
    }
  }

  drawSnake(snake: Snake) {
    const { CELL_SIZE } = CONFIG.GRID;
    snake.body.forEach((seg, i) => {
      this.ctx.fillStyle = i === 0 ? CONFIG.COLORS.SNAKE_HEAD : CONFIG.COLORS.SNAKE_BODY;
      this.ctx.fillRect(
        seg.x * CELL_SIZE + 1,
        seg.y * CELL_SIZE + 1,
        CELL_SIZE - 2,
        CELL_SIZE - 2
      );
    });
  }

  drawFood(food: Food) {
    const { CELL_SIZE } = CONFIG.GRID;
    const t = performance.now() / 1000;
    const pulse = 1 + Math.sin(t * 4) * 0.1;
    const size = (CELL_SIZE - 4) * pulse;

    this.ctx.fillStyle = CONFIG.COLORS.FOOD;
    this.ctx.beginPath();
    this.ctx.arc(
      food.position.x * CELL_SIZE + CELL_SIZE / 2,
      food.position.y * CELL_SIZE + CELL_SIZE / 2,
      size / 2,
      0, Math.PI * 2
    );
    this.ctx.fill();
  }

  drawOverlay(state: GameState) {
    if (state === 'menu') this.drawCenterText('PRESS SPACE TO START');
    else if (state === 'paused') this.drawCenterText('PAUSED');
    else if (state === 'game_over') this.drawCenterText('GAME OVER');
  }

  private drawCenterText(text: string) {
    this.ctx.fillStyle = 'rgba(0,0,0,0.5)';
    this.ctx.fillRect(0, this.canvas.height / 2 - 30, this.canvas.width, 60);
    this.ctx.fillStyle = CONFIG.COLORS.TEXT;
    this.ctx.font = 'bold 20px sans-serif';
    this.ctx.textAlign = 'center';
    this.ctx.fillText(text, this.canvas.width / 2, this.canvas.height / 2 + 7);
  }
}
```

#### `core/Input.ts` —— 输入系统

**职责**：
- 监听键盘 + 触屏
- 防抖与方向缓冲
- 触发回调

```ts
import { CONFIG } from '../config';

export class Input {
  onDirection: ((dir: Direction) => void) | null = null;
  onPause: (() => void) | null = null;
  onConfirm: (() => void) | null = null;
  onReset: (() => void) | null = null;

  constructor(private canvas: HTMLCanvasElement) {
    this.bindKeyboard();
    this.bindTouch();
  }

  private bindKeyboard() {
    window.addEventListener('keydown', (e) => {
      switch (e.key) {
        case 'ArrowUp': case 'w': case 'W':
          this.onDirection?.('UP'); e.preventDefault(); break;
        case 'ArrowDown': case 's': case 'S':
          this.onDirection?.('DOWN'); e.preventDefault(); break;
        case 'ArrowLeft': case 'a': case 'A':
          this.onDirection?.('LEFT'); e.preventDefault(); break;
        case 'ArrowRight': case 'd': case 'D':
          this.onDirection?.('RIGHT'); e.preventDefault(); break;
        case ' ':
          this.onConfirm?.(); e.preventDefault(); break;
        case 'p': case 'P':
          this.onPause?.(); e.preventDefault(); break;
        case 'r': case 'R':
          this.onReset?.(); break;
      }
    });
  }

  private bindTouch() {
    let startX = 0, startY = 0;
    this.canvas.addEventListener('touchstart', (e) => {
      startX = e.touches[0].clientX;
      startY = e.touches[0].clientY;
    }, { passive: true });

    this.canvas.addEventListener('touchend', (e) => {
      const dx = e.changedTouches[0].clientX - startX;
      const dy = e.changedTouches[0].clientY - startY;
      const t = CONFIG.TOUCH.THRESHOLD;

      if (Math.abs(dx) > Math.abs(dy)) {
        if (dx > t) this.onDirection?.('RIGHT');
        else if (dx < -t) this.onDirection?.('LEFT');
      } else {
        if (dy > t) this.onDirection?.('DOWN');
        else if (dy < -t) this.onDirection?.('UP');
      }
    }, { passive: true });

    this.canvas.addEventListener('click', () => this.onConfirm?.());
  }
}

export type Direction = 'UP' | 'DOWN' | 'LEFT' | 'RIGHT';
```

### 3.3 实体层（entities/）

#### `entities/Snake.ts`

```ts
import { CONFIG } from '../config';
import type { Direction } from '../core/Input';

type Point = { x: number; y: number };

export class Snake {
  body: Point[] = [];
  direction: Direction = 'RIGHT';
  nextDirection: Direction = 'RIGHT';
  moveInterval = CONFIG.SPEED.INITIAL_MS;
  tickAccumulator = 0;

  constructor() {
    this.reset();
  }

  reset() {
    const midX = Math.floor(CONFIG.GRID.COLS / 2);
    const midY = Math.floor(CONFIG.GRID.ROWS / 2);
    this.body = [
      { x: midX, y: midY },
      { x: midX - 1, y: midY },
      { x: midX - 2, y: midY },
    ];
    this.direction = 'RIGHT';
    this.nextDirection = 'RIGHT';
    this.moveInterval = CONFIG.SPEED.INITIAL_MS;
    this.tickAccumulator = 0;
  }

  queueDirection(dir: Direction) {
    // 防 180° 反向
    const opposites: Record<Direction, Direction> = {
      UP: 'DOWN', DOWN: 'UP',
      LEFT: 'RIGHT', RIGHT: 'LEFT',
    };
    if (opposites[this.direction] !== dir) {
      this.nextDirection = dir;
    }
  }

  move(food: Point): boolean {
    // 应用缓冲方向
    this.direction = this.nextDirection;
    const head = this.body[0];
    const delta: Record<Direction, Point> = {
      UP: { x: 0, y: -1 },
      DOWN: { x: 0, y: 1 },
      LEFT: { x: -1, y: 0 },
      RIGHT: { x: 1, y: 0 },
    };
    const d = delta[this.direction];
    const newHead: Point = { x: head.x + d.x, y: head.y + d.y };

    this.body.unshift(newHead);

    // 吃到食物？
    if (newHead.x === food.x && newHead.y === food.y) {
      return true; // 不删除尾部
    }

    this.body.pop();
    return false;
  }
}
```

#### `entities/Food.ts`

```ts
import { CONFIG } from '../config';
import { Snake } from './Snake';

type Point = { x: number; y: number };

export class Food {
  position: Point = { x: 0, y: 0 };

  spawn(snake: Snake) {
    const { COLS, ROWS } = CONFIG.GRID;
    const occupied = new Set(snake.body.map((p) => `${p.x},${p.y}`));
    const empty: Point[] = [];

    for (let x = 0; x < COLS; x++) {
      for (let y = 0; y < ROWS; y++) {
        if (!occupied.has(`${x},${y}`)) empty.push({ x, y });
      }
    }

    if (empty.length === 0) return; // 满了
    this.position = empty[Math.floor(Math.random() * empty.length)];
  }
}
```

```ts
import { CONFIG } from '../config';
import { Snake } from '../entities/Snake';

export class CollisionSystem {
  checkWall(snake: Snake): boolean {
    const head = snake.body[0];
    return (
      head.x < 0 || head.x >= CONFIG.GRID.COLS ||
      head.y < 0 || head.y >= CONFIG.GRID.ROWS
    );
  }

  checkSelf(snake: Snake): boolean {
    const [head, ...rest] = snake.body;
    const bodySet = new Set(rest.map((p) => `${p.x},${p.y}`));
    return bodySet.has(`${head.x},${head.y}`);
  }
}
```

```ts
export class AudioSystem {
  private ctx: AudioContext | null = null;

  private getCtx(): AudioContext {
    if (!this.ctx) {
      this.ctx = new (window.AudioContext || (window as any).webkitAudioContext)();
    }
    return this.ctx;
  }

  playEat() {
    this.beep(880, 0.05, 'square');
  }

  playDeath() {
    this.beep(220, 0.3, 'sawtooth');
    setTimeout(() => this.beep(110, 0.3, 'sawtooth'), 100);
  }

  private beep(freq: number, duration: number, type: OscillatorType) {
    try {
      const ctx = this.getCtx();
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.type = type;
      osc.frequency.value = freq;
      gain.gain.setValueAtTime(0.1, ctx.currentTime);
      gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + duration);
      osc.connect(gain).connect(ctx.destination);
      osc.start();
      osc.stop(ctx.currentTime + duration);
    } catch {}
  }
}
```

```ts
export const storage = {
  get<T>(key: string, fallback: T): T {
    try {
      const v = localStorage.getItem(key);
      return v ? (JSON.parse(v) as T) : fallback;
    } catch {
      return fallback;
    }
  },
  set(key: string, value: unknown): void {
    try {
      localStorage.setItem(key, JSON.stringify(value));
    } catch {}
  },
  remove(key: string): void {
    try {
      localStorage.removeItem(key);
    } catch {}
  },
};
```

## 四、数据流图

```
用户操作（键盘/触屏）
       │
       ▼
   Input ──────► 事件回调
       │              │
       │              ▼
       │        Game (主循环)
       │              │
       │              ├──► Snake.move() ──┐
       │              │                    │
       │              ├──► Food.spawn()    │
       │              │                    ├──► CollisionSystem
       │              ├──► ScoreSystem     │       │
       │              │                    │       ▼
       │              └──► AudioSystem     │   撞墙/撞身？
       │                                   │       │ yes
       │                                   │       ▼
       │                                   │   Game.gameOver()
       │                                   │
       ▼                                   │
   UI 更新（DOM）                          ▼
       ▲                              下一帧渲染
       │                                   │
       └──────────── Renderer ◄────────────┘
```

## 五、生命周期

```
main.ts
  └─► new Game(deps)         // 初始化
        ├─ 读取 localStorage
        ├─ 创建实体（Snake, Food）
        ├─ 创建系统（Renderer, Input, ...）
        └─ 绑定输入回调

  └─► game.start()           // 启动主循环
        └─► requestAnimationFrame(loop)
              └─► loop(t)
                    ├─ update(dt)        // 60Hz 逻辑
                    │   ├─ Snake.tick
                    │   ├─ Collision
                    │   ├─ Score
                    │   └─ Audio
                    ├─ render()          // 每帧绘制
                    └─ requestAnimationFrame(loop)
```

## 六、设计模式应用

| 模式 | 应用位置 | 作用 |
| --- | --- | --- |
| **状态机** | `Game.state` | 管理 4 个游戏状态 |
| **观察者** | `Input.onDirection` 回调 | 解耦输入与逻辑 |
| **单例** | `storage`、`random` | 工具类无状态 |
| **实体-系统** | entities + systems | 分离数据与逻辑 |
| **策略模式** | `CONFIG` 配置 | 运行时参数化 |

## 八、测试策略

### 8.1 单元测试（核心算法）
```ts
// snake.test.ts
test('蛇不能 180° 反向', () => {
  const snake = new Snake();
  snake.direction = 'RIGHT';
  snake.queueDirection('LEFT');
  expect(snake.nextDirection).toBe('RIGHT'); // 保持原方向
});

test('食物不在蛇身上生成', () => {
  const snake = new Snake();
  const food = new Food();
  food.spawn(snake);
  expect(snake.body.some(s => s.x === food.position.x && s.y === food.position.y)).toBe(false);
});
```
